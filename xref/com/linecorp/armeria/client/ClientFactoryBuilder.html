49  /**
50   * Builds a new {@link ClientFactory}.
51   *
52   * <h2>Example</h2>
53   * <pre>{@code
54   * final ClientFactory factory = new ClientFactoryBuilder();
55   *     // Set the connection timeout to 5 seconds.
56   *     .connectTimeoutMillis(5000)
57   *     // Set the socket send buffer to 1 MiB.
58   *     .socketOption(ChannelOption.SO_SNDBUF, 1048576)
59   *     // Disable certificate verification; never do this in production!
60   *     .sslContextCustomizer(b -> b.trustManager(InsecureTrustManagerFactory.INSTANCE))
61   *     .build();
62   * }</pre>
63   */
64  public final class ClientFactoryBuilder {
65  
66      private static final KeyedChannelPoolHandler<PoolKey> DEFAULT_CONNECTION_POOL_LISTENER =
67              KeyedChannelPoolHandler.noop();
68  
69      private static final Consumer<SslContextBuilder> DEFAULT_SSL_CONTEXT_CUSTOMIZER = b -> { /* no-op */ };
70  
71      private static final Function<EventLoopGroup, DnsAddressResolverGroup>
72              DEFAULT_ADDRESS_RESOLVER_GROUP_FACTORY = eventLoopGroup -> new DnsAddressResolverGroup(
73              TransportType.datagramChannelType(eventLoopGroup),
74              DnsServerAddressStreamProviders.platformDefault());
75  
76      // Do not accept 1) the options that may break Armeria and 2) the deprecated options.
77      @SuppressWarnings("deprecation")
78      private static final Set<ChannelOption<?>> PROHIBITED_SOCKET_OPTIONS = ImmutableSet.of(
79              ChannelOption.ALLOW_HALF_CLOSURE, ChannelOption.AUTO_READ,
80              ChannelOption.AUTO_CLOSE, ChannelOption.MAX_MESSAGES_PER_READ,
81              ChannelOption.WRITE_BUFFER_HIGH_WATER_MARK, ChannelOption.WRITE_BUFFER_LOW_WATER_MARK,
82              EpollChannelOption.EPOLL_MODE);
83  
84      private static final String MESSAGE_MUTUALLY_EXCLUSIVE =
85              "eventLoopGroup() and useDaemonThreads/numWorkerThreads() are mutually exclusive.";
86  
87      // Netty-related properties:
88      private EventLoopGroup workerGroup = CommonPools.workerGroup();
89      private boolean shutdownWorkerGroupOnClose;
90      private final Map<ChannelOption<?>, Object> socketOptions = new HashMap<>();
91      private Consumer<? super SslContextBuilder> sslContextCustomizer = DEFAULT_SSL_CONTEXT_CUSTOMIZER;
92      private Function<? super EventLoopGroup,
93              ? extends AddressResolverGroup<? extends InetSocketAddress>> addressResolverGroupFactory =
94              DEFAULT_ADDRESS_RESOLVER_GROUP_FACTORY;
95  
96      // Armeria-related properties:
97      private long idleTimeoutMillis = Flags.defaultClientIdleTimeoutMillis();
98      private boolean useHttp2Preface = Flags.defaultUseHttp2Preface();
99      private boolean useHttp1Pipelining = Flags.defaultUseHttp1Pipelining();
100     private KeyedChannelPoolHandler<? super PoolKey> connectionPoolListener = DEFAULT_CONNECTION_POOL_LISTENER;
101 
102     /**
103      * Creates a new instance.
104      */
105     public ClientFactoryBuilder() {
106         connectTimeoutMillis(Flags.defaultConnectTimeoutMillis());
107         socketOption(ChannelOption.SO_KEEPALIVE, true);
108     }
109 
110     /**
111      * Sets the worker {@link EventLoopGroup} which is responsible for performing socket I/O and running
112      * {@link Client#execute(ClientRequestContext, Request)}.
113      * If not set, {@linkplain CommonPools#workerGroup() the common worker group} is used.
114      *
115      * @param shutdownOnClose whether to shut down the worker {@link EventLoopGroup}
116      *                        when the {@link ClientFactory} is closed
117      */
118     public ClientFactoryBuilder workerGroup(EventLoopGroup workerGroup, boolean shutdownOnClose) {
119         this.workerGroup = requireNonNull(workerGroup, "workerGroup");
120         shutdownWorkerGroupOnClose = shutdownOnClose;
121         return this;
122     }
123 
124     /**
125      * Sets the timeout of a socket connection attempt.
126      */
127     public ClientFactoryBuilder connectTimeout(Duration connectTimeout) {
128         requireNonNull(connectTimeout, "connectTimeout");
129         checkArgument(!connectTimeout.isZero() && !connectTimeout.isNegative(),
130                       "connectTimeout: %s (expected: > 0)", connectTimeout);
131         return connectTimeoutMillis(connectTimeout.toMillis());
132     }
133 
134     /**
135      * Sets the timeout of a socket connection attempt in milliseconds.
136      */
137     public ClientFactoryBuilder connectTimeoutMillis(long connectTimeoutMillis) {
138         checkArgument(connectTimeoutMillis > 0,
139                       "connectTimeoutMillis: %s (expected: > 0)", connectTimeoutMillis);
140         return socketOption(ChannelOption.CONNECT_TIMEOUT_MILLIS,
141                             ConvertUtils.safeLongToInt(connectTimeoutMillis));
142     }
143 
144     /**
145      * Sets the options of sockets created by the {@link ClientFactory}.
146      */
147     public <T> ClientFactoryBuilder socketOption(ChannelOption<T> option, T value) {
148         requireNonNull(option, "option");
149         checkArgument(!PROHIBITED_SOCKET_OPTIONS.contains(option),
150                       "prohibited socket option: %s", option);
151 
152         socketOptions.put(option, requireNonNull(value, "value"));
153         return this;
154     }
155 
156     /**
157      * Sets the {@link Consumer} which can arbitrarily configure the {@link SslContextBuilder} that will be
158      * applied to the SSL session. For example, use {@link SslContextBuilder#trustManager} to configure a
159      * custom server CA or {@link SslContextBuilder#keyManager} to configure a client certificate for SSL
160      * authorization.
161      */
162     public ClientFactoryBuilder sslContextCustomizer(Consumer<? super SslContextBuilder> sslContextCustomizer) {
163         this.sslContextCustomizer = requireNonNull(sslContextCustomizer, "sslContextCustomizer");
164         return this;
165     }
166 
167     /**
168      * Sets the factory that creates a {@link AddressResolverGroup} which resolves remote addresses into
169      * {@link InetSocketAddress}es.
170      */
171     public ClientFactoryBuilder addressResolverGroupFactory(
172             Function<? super EventLoopGroup,
173                      ? extends AddressResolverGroup<? extends InetSocketAddress>> addressResolverGroupFactory) {
174         this.addressResolverGroupFactory = requireNonNull(addressResolverGroupFactory,
175                                                           "addressResolverGroupFactory");
176         return this;
177     }
178 
179     /**
180      * Sets the idle timeout of a socket connection. The connection is closed if there is no request in
181      * progress for this amount of time.
182      */
183     public ClientFactoryBuilder idleTimeout(Duration idleTimeout) {
184         requireNonNull(idleTimeout, "idleTimeout");
185         checkArgument(!idleTimeout.isNegative(), "idleTimeout: %s (expected: >= 0)", idleTimeout);
186         return idleTimeoutMillis(idleTimeout.toMillis());
187     }
188 
189     /**
190      * Sets the idle timeout of a socket connection in milliseconds. The connection is closed if there is no
191      * request in progress for this amount of time.
192      */
193     public ClientFactoryBuilder idleTimeoutMillis(long idleTimeoutMillis) {
194         checkArgument(idleTimeoutMillis >= 0, "idleTimeoutMillis: %s (expected: >= 0)", idleTimeoutMillis);
195         this.idleTimeoutMillis = idleTimeoutMillis;
196         return this;
197     }
198 
199     /**
200      * Sets whether to send an HTTP/2 preface string instead of an HTTP/1 upgrade request to negotiate
201      * the protocol version of a cleartext HTTP connection.
202      */
203     public ClientFactoryBuilder useHttp2Preface(boolean useHttp2Preface) {
204         this.useHttp2Preface = useHttp2Preface;
205         return this;
206     }
207 
208     /**
209      * Sets whether to use <a href="https://en.wikipedia.org/wiki/HTTP_pipelining">HTTP pipelining</a> for
210      * HTTP/1 connections. This does not affect HTTP/2 connections. This option is enabled by default.
211      */
212     public ClientFactoryBuilder useHttp1Pipelining(boolean useHttp1Pipelining) {
213         this.useHttp1Pipelining = useHttp1Pipelining;
214         return this;
215     }
216 
217     /**
218      * Sets the listener which is notified on a connection pool event.
219      */
220     public ClientFactoryBuilder connectionPoolListener(
221             KeyedChannelPoolHandler<? super PoolKey> connectionPoolListener) {
222         this.connectionPoolListener = requireNonNull(connectionPoolListener, "connectionPoolListener");
223         return this;
224     }
225 
226     /**
227      * Returns a newly-created {@link ClientFactory} based on the properties of this builder.
228      */
229     public ClientFactory build() {
230         return new DefaultClientFactory(new HttpClientFactory(
231                 workerGroup, shutdownWorkerGroupOnClose, socketOptions, sslContextCustomizer,
232                 addressResolverGroupFactory, idleTimeoutMillis, useHttp2Preface, useHttp1Pipelining,
233                 connectionPoolListener));
234     }
235 
236     @Override
237     public String toString() {
238         return toString(this, workerGroup, shutdownWorkerGroupOnClose, socketOptions,
239                         sslContextCustomizer, addressResolverGroupFactory, idleTimeoutMillis,
240                         useHttp2Preface, useHttp1Pipelining, connectionPoolListener);
241     }
242 
243     static String toString(
244             ClientFactoryBuilder self,
245             EventLoopGroup workerGroup, boolean shutdownWorkerGroupOnClose,
246             Map<ChannelOption<?>, Object> socketOptions,
247             Consumer<? super SslContextBuilder> sslContextCustomizer,
248             Function<? super EventLoopGroup,
249                      ? extends AddressResolverGroup<? extends InetSocketAddress>> addressResolverGroupFactory,
250             long idleTimeoutMillis,
251             boolean useHttp2Preface,
252             boolean useHttp1Pipelining,
253             KeyedChannelPoolHandler<? super PoolKey> connectionPoolListener) {
254 
255         final ToStringHelper helper = MoreObjects.toStringHelper(self);
256         helper.add("workerGroup", workerGroup + " (shutdownOnClose=" + shutdownWorkerGroupOnClose + ')')
257               .add("socketOptions", socketOptions)
258               .add("idleTimeoutMillis", idleTimeoutMillis)
259               .add("useHttp2Preface", useHttp2Preface)
260               .add("useHttp1Pipelining", useHttp1Pipelining);
261 
262         if (connectionPoolListener != DEFAULT_CONNECTION_POOL_LISTENER) {
263             helper.add("connectionPoolListener", connectionPoolListener);
264         }
265 
266         if (sslContextCustomizer != DEFAULT_SSL_CONTEXT_CUSTOMIZER) {
267             helper.add("sslContextCustomizer", sslContextCustomizer);
268         }
269 
270         if (addressResolverGroupFactory != DEFAULT_ADDRESS_RESOLVER_GROUP_FACTORY) {
271             helper.add("addressResolverGroupFactory", addressResolverGroupFactory);
272         }
273 
274         return helper.toString();
275     }
